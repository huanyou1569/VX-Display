import os
out_dir = r"C:\Users\1569\Desktop\stm32\vx-display\fbx_pipeline\frames_stl_valid"
os.makedirs(out_dir, exist_ok=True)

start = 0  # ← 改成实际起始帧
end = 21     # ← 改成实际结束帧

scene = bpy.context.scene
for i, fno in enumerate(range(start, end + 1)):
    scene.frame_set(fno)
    bpy.ops.object.select_all(action="DESELECT")
    for o in bpy.data.objects:
        if o.type == "MESH":
            o.select_set(True)
    path = os.path.join(out_dir, f"frame_{i:03d}.stl")
    bpy.ops.export_mesh.stl(filepath=path, use_selection=True, global_scale=1.0)
    print(f"[{i+1}/{end-start+1}] 帧 {fno} → {path}")

print("完成")
